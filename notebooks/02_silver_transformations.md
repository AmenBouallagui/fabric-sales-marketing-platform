# 02 Silver Transformations Notebook Plan

## Objective

Design a future Microsoft Fabric notebook that reads Bronze Delta tables, standardizes and type-casts source records, deduplicates by business key, applies business validations, and writes cleaned Silver Delta tables for downstream Gold modeling.

This Markdown file is a notebook implementation plan. It is not an executable `.ipynb` artifact yet.

## Parameters

Planned notebook parameters:

- `silver_processed_at`: run timestamp, generated by the notebook or passed by a pipeline.
- `fail_on_missing_bronze_table`: boolean flag, default `true`.
- `write_mode`: default `overwrite` for MVP development; future production mode may use merge/upsert.
- `dq_reject_mode`: default `mark`, with a future option to write rejected rows to dedicated tables.

## Standardize Columns

String cleanup should happen before type casting and validation.

```python
def trim_string_columns(df):
    for column_name, dtype in df.dtypes:
        if dtype == "string":
            df = df.withColumn(column_name, F.trim(F.col(column_name)))
    return df


def normalize_categories(df, entity_name: str):
    if entity_name == "orders":
        df = df.withColumn("payment_status", F.initcap(F.lower(F.col("payment_status"))))
    if entity_name == "customers":
        df = df.withColumn("status", F.initcap(F.lower(F.col("status"))))
        df = df.withColumn("customer_segment", F.initcap(F.lower(F.col("customer_segment"))))
    if entity_name == "campaigns":
        df = df.withColumn("channel", F.initcap(F.lower(F.col("channel"))))
    return df
```

def cast_common_metadata(df):
    return (
        df
        .withColumn("source_updated_at", F.to_timestamp("source_updated_at"))
        .withColumn("ingested_at", F.to_timestamp("ingested_at"))

def cast_entity_types(df, entity_name: str):
    if entity_name == "orders":
        return (
            df
            .withColumn("order_date", F.to_date("order_date"))
            .withColumn("quantity", F.col("quantity").cast(IntegerType()))
            .withColumn("unit_price", F.col("unit_price").cast(DecimalType(18, 2)))
            .withColumn("discount_amount", F.col("discount_amount").cast(DecimalType(18, 2)))

            .withColumn("total_amount", F.col("total_amount").cast(DecimalType(18, 2)))
            .withColumn("refund_flag", F.col("refund_flag").cast(BooleanType()))

## Deduplicate With Window Functions

Silver current records should be selected with deterministic latest-record logic.

```python
def deduplicate_current_records(df, business_key: str):
    window_spec = (
        Window
        .partitionBy(business_key)
        .orderBy(
            F.col("source_updated_at").desc_nulls_last(),
            F.col("ingested_at").desc_nulls_last(),
            F.col("row_hash").desc_nulls_last(),
        )
    )

    return (
        df
        .withColumn("record_rank", F.row_number().over(window_spec))
        .withColumn("is_current_record", F.col("record_rank") == F.lit(1))
        .filter(F.col("record_rank") == 1)
        .drop("record_rank")
    )
```

## Apply Business Validations

Record-level data quality flags should be calculated before writing Silver outputs.

```python
def add_dq_status(df, entity_name: str):
    issue_reasons = []

    if entity_name == "customers":
        issue_reasons = [
            F.when(F.col("customer_id").isNull(), F.lit("missing customer_id")),
            F.when(F.col("customer_name").isNull(), F.lit("missing customer_name")),
            F.when(F.col("signup_date").isNull(), F.lit("missing signup_date")),
        ]

    issue_array = F.array(*[reason for reason in issue_reasons])
    issue_text = F.concat_ws("; ", F.expr("filter(dq_issues, x -> x is not null)"))

    return (
        df
        .withColumn("dq_issues", issue_array)
        .withColumn("dq_issue_reason", issue_text)
        .withColumn(
            "dq_status",
            F.when(F.length(F.col("dq_issue_reason")) > 0, F.lit("rejected")).otherwise(F.lit("valid")),
        )
        .drop("dq_issues")
    )
```

The production implementation should include entity-specific checks for monetary values, valid categorical values, required fields, and referential integrity.

## Referential Integrity Checks

After base Silver tables are built, dependent entities should be checked against their parent entities.

```python
def mark_order_referential_failures(orders_df, customers_df, products_df, campaigns_df):
    valid_customers = customers_df.select("customer_id").distinct()
    valid_products = products_df.select("product_id").distinct()
    valid_campaigns = campaigns_df.select("campaign_id").distinct()

    checked = (
        orders_df
        .join(valid_customers.withColumn("valid_customer", F.lit(True)), "customer_id", "left")
        .join(valid_products.withColumn("valid_product", F.lit(True)), "product_id", "left")
        .join(valid_campaigns.withColumn("valid_campaign", F.lit(True)), "campaign_id", "left")
    )

    return checked.withColumn(
        "dq_status",
        F.when(F.col("valid_customer").isNull(), F.lit("rejected"))
        .when(F.col("valid_product").isNull(), F.lit("rejected"))
        .when(F.col("campaign_id").isNotNull() & F.col("valid_campaign").isNull(), F.lit("rejected"))
        .otherwise(F.col("dq_status")),
    )
```

## Write Silver Delta Tables

Silver tables can use overwrite semantics during MVP development.

```python
def write_silver_table(df, table_name: str, write_mode: str = "overwrite"):
    (
        df.write
        .format("delta")
        .mode(write_mode)
        .option("overwriteSchema", "true")
        .saveAsTable(table_name)
    )
```
